#include<stdio.h>
#include<windows.h>
int n;
int a[10][10];
int c[10];
int number=0,max=0;

int fsd=0;
int canplace(int row,int col)
{

	int i,j;
	printf("\n\n******************  µÚ%d´Î  *************\n\n",fsd++);
	if(fsd%10==0)
		system("pause");
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
			printf("%4d",a[i][j]);
		printf("\n");
	}
	int flag1=0,flag2=0,r;
		r=row;
	if(a[row][col]==-1)
		return 0;
	while(row>=0 && a[row][col]==0)
		row--;
	if(a[row][col]==-1 || row==-1)
		flag1=1;

	while(col>=0 && a[r][col]==0)
		col--;
	if(a[r][col]==-1 || col==-1)
		flag2=1;
	if(flag1&&flag2)
		return 1;
	else return 0;

}
void search(int m)
{
	int r,c;
	r=m/n;
	c=m%n;
	if(m>=n*n)
	{
		if(number>max)
			max=number;
	}
	else
	{

		search(m+1);
		if(canplace(r,c))
		{
			a[r][c]=1;
			number++;
			search(m+1);
			a[r][c]=0;
			number--;
		}


	}
}

void main()
{
	int i,j,k=0;
	char str[10];
	scanf("%d",&n);
	getchar();
	while(n!=0)
	{
		for(i=0;i<n;i++)
		{
			gets(str);
			for(j=0;j<n;j++)
			{
				if(str[j]=='.')
					a[i][j]=0;
				else
					a[i][j]=-1;
			}

		}
		search(0);
		c[k++]=max;
		max=0;
		number=0;
		scanf("%d",&n);
		getchar();
	}
	for(i=0;i<k;i++)
		printf("%d\n",c[i]);
}
